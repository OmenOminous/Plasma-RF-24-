return function(arg1)
	_G.distance = arg1
    local HttpService = game:GetService("HttpService")
    local gistUrl = "https://gist.githubusercontent.com/SigmaSploiter1/1d4cc159218ad8ffd4c8b561049c04c4/raw/73e6a1fbdb5cd9c67904b9713b19e80de97fd428/gistfile1.txt"
    
    function isWhitelisted(hwid)
        local response = game:HttpGet(gistUrl)
        local data = HttpService:JSONDecode(response)
        for _, v in pairs(data.whitelisted_hwids) do
            if v == hwid then
                return true
            end
        end
        return false
    end
    
    local hwid = game:GetService("RbxAnalyticsService"):GetClientId()
    
    if isWhitelisted(hwid) then
    	game:GetService("StarterGui"):SetCore("SendNotification",{
       		Title = "WHITELIST",
        	Text = "HWID IS WHITELISTED, EXECUTING SCRIPT NOW!", 
        	Button1 = "Okay!",
        	Duration = 60
    	})
    	loadstring(game:HttpGet("https://raw.githubusercontent.com/OmenOminous/Plasma-RF-24-/main/Plasma.txt"))()
    else
    	game:GetService("StarterGui"):SetCore("SendNotification",{
       		Title = "WHITELIST",
        	Text = "HWID ISN'T WHITELISTED, WHAT ARE YOU DOING?"
    	})
    end
end
